#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct member {
	int age;
	char name[100];
	int time;
};

struct member M[100000];

void inputM(int N);
void printM(int N);

int main() {
	int N;
	scanf("%d", &N);

	inputM(N);
	printM(N);
}

void inputM(int N) {
	int i, j, k, min;
	struct member temp;
	for (i = 0; i < N; i++) {
		scanf("%d", &M[i].age);
		scanf("%s", &M[i].name);
		M[i].time = i;

		for (j = 0; j < i; j++) {
			if (M[i].age < M[j].age) {
				break;
			}
			else if (M[i].age == M[j].age) {
				if (M[i].time < M[j].time)
					break;
			}
		}

		temp = M[i];
		for (k = i; k > j; k--) {
			M[k] = M[k-1];
		}
		M[j] = temp;
	}
}

void printM(int N) {
	int i;
	for (i = 0; i < N; i++) {
		printf("%d ", M[i].age);
		printf("%s\n", M[i].name);
	}
}
