#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct member {
	int age;
	char name[100];
	int time;
};

struct member M[100000];

void inputM(int N);
void sortM(int N);
void printM(int N);

int main() {
	int N;
	scanf("%d", &N);

	inputM(N);
	sortM(N);
	printM(N);
}

void inputM(int N) {
	int i;
	for (i = 0; i < N; i++) {
		scanf("%d", &M[i].age);
		scanf("%s", &M[i].name);
		M[i].time = i;
	}
}

void printM(int N) {
	int i;
	for (i = 0; i < N; i++) {
		printf("%d ", M[i].age);
		printf("%s\n", M[i].name);
	}
}

void sortM(int N) {
	int i, j, min;
	struct member temp;
	for (i = 0; i < N-1; i++) {
		min = i;
		for (j = i+1; j < N; j++) {
			if (M[min].age > M[j].age) {
				min = j;
			}
			else if (M[min].age == M[j].age) {
				if (M[min].time > M[j].time)
					min = j;
			}
		}

		temp = M[min];
		M[min] = M[i];
		M[i] = temp;
	}
}
